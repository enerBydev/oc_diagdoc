    .select('quiz_intentos')
    .eq('solicitud_id', solicitudId)

  // Actualizar
  await supabase
    .from('capacitacion_progreso')
    .update({
      quiz_intentos: intentos,
      quiz_score: score,

    .eq('solicitud_id', solicitudId);
  
  if (score >= minScore) {
    // Aprobar y continuar
    await supabase
      .from('operador_solicitudes')
      .update({ estado: 'capacitacion_aprobada' })
