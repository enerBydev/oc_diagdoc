# 3.3.1.1 Schemas Zod

> Schemas de validación con Zod para todos los formularios.

---

## Ejemplo: Schema Servicio

```typescript
import { z } from 'zod';

export const servicioSchema = z.object({
  codigo: z.string().min(3).max(100),
  nombre: z.string().min(3).max(200),
  servicio_minutos: z.number().min(10).max(480),
  buffer_minutos: z.number().min(0).max(60).default(15),
  precio_base: z.number().min(50),
  descripcion: z.string().optional(),
});

export const usuarioSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8).regex(/[A-Z]/, "Requires Uppercase").regex(/[0-9]/, "Requires Number"),
  nombre_completo: z.string().min(5),
  telefono: z.string().regex(/^\+52\d{10}$/, "Format: +52XXXXXXXXXX"),
  role: z.enum(['admin', 'operador', 'client_b2c', 'client_b2b'])
});

export const vehiculoSchema = z.object({
  placa: z.string().regex(/^[A-Z0-9-]{6,9}$/, "Placa inválida").toUpperCase(),
  marca: z.string().min(2),
  modelo: z.string().min(2),
  anio: z.number().min(1950).max(new Date().getFullYear() + 1),
  color: z.string().min(3),
  tipo: z.enum(['sedan', 'suv', 'pickup', 'moto'])
});

export const citaSchema = z.object({
  cliente_id: z.string().uuid(),
  servicios_ids: z.array(z.string().uuid()).min(1),
  fecha_hora: z.date().refine(date => date > new Date(), "La fecha debe ser futura"),
  ubicacion: z.object({
    lat: z.number(),
    lng: z.number(),
    direccion: z.string()
  }),
  metodo_pago_id: z.string().uuid()
});
```
