## LÃ³gica de Monitoreo

```typescript
// Ejecutar despuÃ©s de cada servicio completado
const monitorearOperadorNuevo = async (operadorId: string, nuevoRating: number) => {
  const config = await getRemoteConfig('periodo_prueba');
  
  const { data: operador } = await supabase
    .from('operadores')
    .select('servicios_completados, rating_promedio, consolidado')
    .eq('id', operadorId)
    .single();
  
  if (operador.consolidado) return; // Ya consolidado
  
  // Actualizar contador
  const nuevosServicios = operador.servicios_completados + 1;
  const nuevoPromedio = calcularNuevoPromedio(
    operador.rating_promedio, 
    operador.servicios_completados,
    nuevoRating
  );
  
  await supabase
    .from('operadores')
    .update({
      servicios_completados: nuevosServicios,
      rating_promedio: nuevoPromedio
    })
    .eq('id', operadorId);
  
  // Verificar alertas
  if (nuevoPromedio < 3.0) {
    await suspenderOperador(operadorId, 'rating_muy_bajo');
    await notifyAdmin('OPERADOR_SUSPENDIDO', { operadorId, rating: nuevoPromedio });
    return;
  }
  
  if (nuevoPromedio < 3.5) {
    await notifyAdmin('OPERADOR_ALERTA', { operadorId, rating: nuevoPromedio });
  }
  
  // Verificar consolidaciÃ³n
  if (nuevosServicios >= config.servicios_requeridos && nuevoPromedio >= config.rating_minimo) {
    await consolidarOperador(operadorId);
  }
};
```
