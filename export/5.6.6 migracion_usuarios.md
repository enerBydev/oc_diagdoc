// POST /api/admin/usuarios/migrar
export default defineEventHandler(async (event) => {
  requireRole(event, ['admin']);
  
  const body = await readBody(event);
  const { userId, empresaId, emailCorporativo, razon } = body;
  
  // Validar empresa tiene contrato activo
  const empresa = await db
    .select()
    .from(empresas_b2b)
    .where(and(
      eq(empresas_b2b.id, empresaId),
      eq(empresas_b2b.contrato_activo, true)
    ))
    .first();
  
  if (!empresa) {
    throw createError({
      statusCode: 400,
      message: 'Empresa no tiene contrato activo'
    });
  }
  
  // Ejecutar migraci√≥n
  await db.transaction(async (tx) => {
    // Actualizar usuario
    await tx
      .update(users)
      .set({
        tipo_usuario: 'corporate_plus',
        empresa_id: empresaId,
        email_secundario: emailCorporativo || null,
        migrado_at: new Date()
      })
      .where(eq(users.id, userId));
    
    // Audit log
    await tx.insert(audit_logs).values({
      accion: 'MIGRACION_USUARIO',
      admin_id: event.context.user.id,
      usuario_id: userId,
      empresa_id: empresaId,
      razon,
      detalles: { tipo_anterior: 'b2c_puro', tipo_nuevo: 'corporate_plus' }
    });
  });
  
  // Notificar usuario
  await sendNotification({
    userId,
    tipo: 'migracion_corporate',
    mensaje: `¬°Bienvenido a Corporate+! Ahora tienes ${empresa.nombre} como empresa.`
  });
  
  return { success: true };
