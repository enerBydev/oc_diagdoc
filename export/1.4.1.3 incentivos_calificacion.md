## Modelo de Datos

```sql
-- Puntos del cliente
CREATE TABLE cliente_puntos (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cliente_id UUID REFERENCES users(id) UNIQUE,
  
  puntos_totales INTEGER DEFAULT 0,
  puntos_disponibles INTEGER DEFAULT 0,
  puntos_canjeados INTEGER DEFAULT 0,
  
  racha_actual INTEGER DEFAULT 0,
  mejor_racha INTEGER DEFAULT 0,
  
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Historial de puntos
CREATE TABLE puntos_historial (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cliente_id UUID REFERENCES users(id),
  
  tipo VARCHAR(30), -- 'calificacion', 'comentario', 'testimonio', 'bonus_racha', 'canje'
  puntos INTEGER, -- positivo = ganados, negativo = canjeados
  descripcion TEXT,
  
  referencia_id UUID, -- servicio_id o cupon_id
  
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Badges desbloqueados
CREATE TABLE cliente_badges (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cliente_id UUID REFERENCES users(id),
  badge_id VARCHAR(30), -- 'critico_estrella', 'racha_5', etc.
  
  desbloqueado_at TIMESTAMPTZ DEFAULT now(),
  
  UNIQUE(cliente_id, badge_id)
);
```
