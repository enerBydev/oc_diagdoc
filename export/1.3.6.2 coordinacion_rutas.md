## Actualizaci√≥n en Tiempo Real

```typescript

const actualizarUbicacion = async (operadorId: string, lat: number, lng: number) => {
  await supabase
    .from('ubicaciones_operador')
    .upsert({
      operador_id: operadorId,
      lat,
      lng,
      updated_at: new Date()
    });
  
  // Broadcast a Admin
  await supabase.channel('ubicaciones').send({
    type: 'broadcast',
    event: 'ubicacion_update',
    payload: { operadorId, lat, lng }
  });
};
```
