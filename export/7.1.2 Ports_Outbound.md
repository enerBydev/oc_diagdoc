# 7.1.2 Puertos Outbound (Secondary Ports)

> Interfaces que el dominio necesita pero NO implementa.

---

## Concepto

Los **puertos outbound** son interfaces que representan las dependencias externas del dominio (base de datos, APIs externas, servicios).

---

## Interfaces OnlyCar

```typescript
// domain/ports/outbound/ICitaRepository.ts
interface ICitaRepository {
  findById(id: string): Promise<Cita | null>;
  findByOperador(operadorId: string): Promise<Cita[]>;
  findByCliente(clienteId: string): Promise<Cita[]>;
  save(cita: Cita): Promise<void>;
  delete(id: string): Promise<void>;
}

// domain/ports/outbound/IPagoGateway.ts
interface IPagoGateway {
  createPaymentIntent(monto: number, metadata: any): Promise<PaymentIntent>;
  confirmPayment(intentId: string): Promise<PaymentResult>;
  refund(paymentId: string, amount?: number): Promise<RefundResult>;
}

// domain/ports/outbound/INotificacionService.ts
interface INotificacionService {
  sendPush(userId: string, message: NotificationPayload): Promise<void>;
  sendEmail(email: string, template: string, data: any): Promise<void>;
  sendSMS(phone: string, message: string): Promise<void>;
}

// domain/ports/outbound/IUbicacionService.ts
interface IUbicacionService {
  getCurrentPosition(): Promise<Coordenadas>;
  watchPosition(callback: (pos: Coordenadas) => void): () => void;
  calculateDistance(from: Coordenadas, to: Coordenadas): number;
}
```
