## Modelo de Datos

```sql
CREATE TABLE materiales_catalogo (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  nombre VARCHAR(100) NOT NULL,
  unidad VARCHAR(20) NOT NULL,  -- litro, pieza, gramo
  costo_unidad DECIMAL(10,2) NOT NULL,
  proveedor VARCHAR(100),
  marca VARCHAR(50),
  activo BOOLEAN DEFAULT true,
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE servicio_materiales (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  servicio_id VARCHAR(50) NOT NULL,
  material_id UUID REFERENCES materiales_catalogo(id),
  cantidad_uso DECIMAL(10,4) NOT NULL,  -- cantidad por servicio
  costo_calculado DECIMAL(10,2) GENERATED ALWAYS AS 
    (cantidad_uso * (SELECT costo_unidad FROM materiales_catalogo WHERE id = material_id)) STORED
);
```
