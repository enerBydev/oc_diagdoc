## Modelo de Datos

```sql
CREATE TABLE operador_documentos (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  solicitud_id UUID NOT NULL REFERENCES operador_solicitudes(id),
  tipo VARCHAR(30) NOT NULL,  -- ine_frente, ine_reverso, curp, domicilio, selfie
  archivo_url TEXT NOT NULL,
  archivo_nombre VARCHAR(255),
  archivo_size INT,
  estado VARCHAR(20) DEFAULT 'pendiente',  -- pendiente, aprobado, rechazado
  motivo_rechazo TEXT,
  verificado_por UUID REFERENCES users(id),
  verificado_at TIMESTAMPTZ,
  metadata JSONB,  -- { curp: "...", nombre: "...", vigencia: "..." }
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE UNIQUE INDEX idx_doc_tipo_solicitud 
  ON operador_documentos(solicitud_id, tipo);
```
