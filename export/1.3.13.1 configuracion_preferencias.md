```typescript
// composables/useReminderPreferences.ts
export const useReminderPreferences = () => {
  const supabase = useSupabaseClient()
  const user = useSupabaseUser()
  
  const preferences = ref<ReminderPreferences | null>(null)
  
  const fetchPreferences = async () => {
    const { data } = await supabase
      .from('reminder_preferences')
      .select('*')
      .eq('cliente_id', user.value?.id)
      .single()
    
    preferences.value = data
  }
  
  const updatePreferences = async (updates: Partial<ReminderPreferences>) => {
    await supabase
      .from('reminder_preferences')
      .update(updates)
      .eq('cliente_id', user.value?.id)
  }
  
  const optOut = async () => {
    await updatePreferences({ activo: false })
  }
  
  return { preferences, fetchPreferences, updatePreferences, optOut }
}
```
