## Modelo de Datos

```sql
CREATE TABLE calificaciones_operador (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  calificacion_id UUID REFERENCES calificaciones(id),
  operador_id UUID REFERENCES users(id),
  servicio_id UUID REFERENCES servicios(id),
  
  -- Dimensiones (1-5)
  puntualidad SMALLINT CHECK (puntualidad BETWEEN 1 AND 5),
  profesionalismo SMALLINT CHECK (profesionalismo BETWEEN 1 AND 5),
  calidad SMALLINT CHECK (calidad BETWEEN 1 AND 5),
  comunicacion SMALLINT CHECK (comunicacion BETWEEN 1 AND 5),
  
  -- Score ponderado calculado
```

```typescript
const calcularScorePonderado = (ratings: DimensionRatings): number => {
  const pesos = {
    puntualidad: 0.25,
    profesionalismo: 0.25,
    calidad: 0.30,
    comunicacion: 0.20,
  };
  
  return Object.entries(ratings).reduce(
    (total, [dim, valor]) => total + (valor * pesos[dim as keyof typeof pesos]),
    0
```
