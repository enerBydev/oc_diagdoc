## Cรกlculo de Score Total

```typescript
const calcularScoreTotal = (respuestas: Respuestas): ScoreResult => {
  const experiencia = calcularSeccion(respuestas, 'experiencia');
  const disponibilidad = calcularSeccion(respuestas, 'disponibilidad');
  const motivacion = calcularSeccion(respuestas, 'motivacion');
  const situaciones = calcularSeccion(respuestas, 'situaciones');
  
  const total = 
    (experiencia * 0.30) +
    (disponibilidad * 0.25) +
    (motivacion * 0.25) +
    (situaciones * 0.20);
  
  return {
    experiencia: Math.round(experiencia),
    disponibilidad: Math.round(disponibilidad),
    motivacion: Math.round(motivacion),
    situaciones: Math.round(situaciones),
    total: Math.round(total),
    aprobado: total >= 60
  };
};
```
