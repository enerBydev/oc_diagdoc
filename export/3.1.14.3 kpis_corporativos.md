```typescript
interface KPIsCorporativos {
  runway: number;
  liquidez: number;
  margenOperativo: number;
  roe: number;
  crecimientoMoM: number;
}

function calcularKPIs(datos: DatosFinancieros): KPIsCorporativos {
  return {
    runway: datos.reservas / datos.gastosMensuales,
    liquidez: datos.activosCorrientes / datos.pasivosCorrientes,
    margenOperativo: (datos.ingresos - datos.costos) / datos.ingresos * 100,
    roe: datos.utilidadNeta / datos.capitalSocial * 100,
    crecimientoMoM: (datos.ingresoActual - datos.ingresoAnterior) / datos.ingresoAnterior * 100
  };
}
```
