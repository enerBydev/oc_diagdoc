## Composable

```typescript
// composables/useAdminChatHub.ts
export const useAdminChatHub = () => {
  const filtros = ref({
    tipoUsuario: 'todos', // todos, operador, b2c, corporate, b2b
    urgencia: 'todas',
    estado: 'todos',
    fecha: 'hoy'
  });
  
  const conversaciones = computed(() => {
    return allConversations.value
      .filter(c => aplicarFiltros(c, filtros.value))
      .sort((a, b) => ordenarPorUrgencia(a, b));
  });
  
  const contadores = computed(() => ({
    todos: allConversations.value.length,
    operador: allConversations.value.filter(c => c.tipo === 'operador').length,
    b2c: allConversations.value.filter(c => c.tipo === 'b2c').length,
    corporate: allConversations.value.filter(c => c.tipo === 'corporate_plus').length,
    b2b: allConversations.value.filter(c => c.tipo === 'b2b').length
  }));
  
  return { filtros, conversaciones, contadores };
};
```
