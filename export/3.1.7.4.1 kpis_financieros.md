## API: Obtener KPIs

```typescript
// server/api/admin/kpis/financieros.get.ts
export default defineEventHandler(async () => {
  const inicioMes = startOfMonth(new Date());
  const finMes = endOfMonth(new Date());
  
  const { data: servicios } = await supabase
    .from('citas')
    .select('precio_total, comision_operador')
    .gte('completado_at', inicioMes)
    .lte('completado_at', finMes)
    .eq('estado', 'completado');
  
  const ingresoTotal = servicios.reduce((s, c) => s + c.precio_total, 0);
  const comisionesTotal = servicios.reduce((s, c) => s + c.comision_operador, 0);
  const utilidad = ingresoTotal - comisionesTotal - (ingresoTotal * 0.131);
  
  return {
    ingreso: ingresoTotal,
    utilidad,
    margen: (utilidad / ingresoTotal) * 100,
    servicios: servicios.length,
    ticketPromedio: ingresoTotal / servicios.length
  };
});
```
