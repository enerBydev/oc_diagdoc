## Consultas SQL

```sql
-- Incidencias por categor√≠a este mes
SELECT 
  categoria,
  COUNT(*) as total,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) as porcentaje
FROM incidencias
WHERE created_at >= date_trunc('month', now())
GROUP BY categoria
ORDER BY total DESC;

-- Operadores con mayor tasa de incidencias
SELECT 
  u.nombre,
  COUNT(i.id) as incidencias,
  COUNT(DISTINCT s.id) as servicios,
  ROUND(COUNT(i.id) * 100.0 / COUNT(DISTINCT s.id), 2) as tasa
FROM users u
LEFT JOIN servicios s ON s.operador_id = u.id
LEFT JOIN incidencias i ON i.cita_id = s.cita_id
WHERE u.rol = 'operador'
  AND s.created_at >= date_trunc('month', now())
GROUP BY u.id, u.nombre
HAVING COUNT(DISTINCT s.id) > 10
ORDER BY tasa DESC;

-- Tiempo promedio de respuesta
SELECT 
  AVG(EXTRACT(EPOCH FROM (en_revision_at - created_at)) / 60) as minutos_promedio
FROM incidencias
WHERE en_revision_at IS NOT NULL
  AND created_at >= date_trunc('month', now());
```
