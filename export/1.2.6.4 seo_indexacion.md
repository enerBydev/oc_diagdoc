## Meta Tags por Página

```typescript
// nuxt.config.ts
export default defineNuxtConfig({
  app: {
    head: {
      titleTemplate: '%s | OnlyCar',
      meta: [
        { name: 'robots', content: 'index, follow' },
        { property: 'og:site_name', content: 'OnlyCar' },
      ],
    },
  },
});

// pages/servicios/index.vue
useHead({
  title: 'Servicios de Limpieza Vehicular',
  meta: [
    { 
      name: 'description', 
      content: 'Lavado express, detailing premium, limpieza de interiores. Servicio a domicilio en Monterrey y área metropolitana.' 
    },
    { property: 'og:title', content: 'Servicios de Limpieza Vehicular | OnlyCar' },
    { property: 'og:description', content: 'Agenda tu lavado de auto a domicilio. Operadores certificados, productos premium.' },
    { property: 'og:image', content: '/og/servicios.jpg' },
  ],
});
```
