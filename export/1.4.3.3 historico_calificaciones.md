// Detectar tendencia
const detectarTendencia = (scores: number[]): 'mejorando' | 'estable' | 'bajando' => {
  if (scores.length < 2) return 'estable';
  
  const recientes = scores.slice(-3);
  const promedio = recientes.reduce((a, b) => a + b, 0) / recientes.length;
  const primero = recientes[0];
  
  const cambio = promedio - primero;
  
  if (cambio > 5) return 'mejorando';
  if (cambio < -5) return 'bajando';
  return 'estable';
};
