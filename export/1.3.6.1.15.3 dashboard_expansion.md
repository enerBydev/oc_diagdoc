## Vista SQL para Dashboard

```sql
CREATE VIEW v_dashboard_ciudades AS
SELECT
  c.id,
  c.nombre,
  e.nombre as estado_nombre,
  c.estado,
  c.created_at,
  c.admin_regional_id,
  u.nombre as admin_nombre,
  
  -- Operadores
  (SELECT COUNT(*) FROM operadores o WHERE o.ciudad_id = c.id AND o.estado = 'activo') as operadores_activos,
  
  -- Servicios √∫ltimos 30 d√≠as
  (SELECT COUNT(*) FROM citas ci 
   JOIN operadores o ON ci.operador_id = o.id 
   WHERE o.ciudad_id = c.id AND ci.created_at > NOW() - INTERVAL '30 days') as servicios_mes,
  
  -- Rating promedio
  (SELECT COALESCE(AVG(ci.rating_cliente), 0) 
   FROM citas ci 
   JOIN operadores o ON ci.operador_id = o.id 
   WHERE o.ciudad_id = c.id AND ci.estado = 'completada') as rating_promedio,
  
  -- Zonas
  (SELECT COUNT(*) FROM zonas z WHERE z.ciudad_id = c.id AND z.activa = true) as zonas_activas

FROM ciudades c
JOIN estados e ON c.estado_id = e.id
LEFT JOIN users u ON c.admin_regional_id = u.id
ORDER BY c.estado, c.nombre;
```
