## Vista: Requisitos por Servicio

```sql
CREATE VIEW v_requisitos_servicio AS
SELECT
  s.id as servicio_id,
  s.nombre as servicio_nombre,
  th.id as herramienta_tipo_id,
  th.nombre as herramienta_nombre,
  ch.nombre as categoria,
  rs.obligatorio
FROM servicios s
JOIN requisitos_servicio rs ON s.id = rs.servicio_id
JOIN tipos_herramientas th ON rs.tipo_herramienta_id = th.id
JOIN categorias_herramientas ch ON th.categoria_id = ch.id
ORDER BY s.nombre, ch.orden, th.nombre;
```
