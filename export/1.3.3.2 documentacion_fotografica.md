## Procesamiento

```typescript
const procesarFoto = async (file: File, servicioId: string, tipo: 'antes' | 'despues') => {
  // Convertir HEIC si necesario
  const jpeg = file.type === 'image/heic' 
    ? await heicToJpeg(file) 
    : file;
  
  // Redimensionar
  const resized = await resizeImage(jpeg, { maxWidth: 2048, maxHeight: 2048 });
  
  // Comprimir
  const compressed = await compressImage(resized, { quality: 0.8 });
  
  // Generar thumbnail
  const thumb = await resizeImage(compressed, { maxWidth: 400, maxHeight: 400 });
  
  // Subir a R2
  const uuid = crypto.randomUUID();
  const pathFull = `servicios/${servicioId}/${tipo}/${uuid}_full.jpg`;
  const pathThumb = `servicios/${servicioId}/${tipo}/${uuid}_thumb.jpg`;
  
  await r2.put(pathFull, compressed);
  await r2.put(pathThumb, thumb);
  
  // Registrar en BD
  await db.insert(fotos_servicio).values({
    id: uuid,
    servicio_id: servicioId,
    tipo,
    url_full: `${R2_PUBLIC_URL}/${pathFull}`,
    url_thumb: `${R2_PUBLIC_URL}/${pathThumb}`,
    metadata: { ancho: resized.width, alto: resized.height, tamanio: compressed.size }
  });
};
```
