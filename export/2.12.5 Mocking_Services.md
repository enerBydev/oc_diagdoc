## Mock Service Worker (MSW)

Intercepta peticiones a nivel de red (Service Worker) y retorna respuestas mockeadas.

```typescript
// mocks/handlers.ts
import { http, HttpResponse } from 'msw';

export const handlers = [
  // Mock Stripe Payment
  http.post('https://api.stripe.com/v1/payment_intents', () => {
    return HttpResponse.json({
      id: 'pi_mock_123',
      status: 'succeeded'
    });
  }),
  
  // Mock User Profile
  http.get('/api/user', () => {
    return HttpResponse.json({
      id: 'user_mock',
      name: 'Test Operator',
      role: 'operador'
    });
  })
];
```
