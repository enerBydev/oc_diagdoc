## Slash Commands

Escribir `/` en el input activa autocompletado:

```typescript
const slashCommands = [
  { cmd: '/saludo', template: 'saludo' },
  { cmd: '/confirmar', template: 'confirmacion' },
  { cmd: '/seguimiento', template: 'seguimiento' },
  { cmd: '/gracias', template: 'agradecimiento' },
  { cmd: '/disculpa', template: 'disculpa' },
  { cmd: '/propuesta', action: 'abrirModalPropuesta' },
  { cmd: '/contrato', action: 'abrirModalContrato' }
];

const handleSlashCommand = (input: string) => {
  const command = slashCommands.find(c => input.startsWith(c.cmd));
  if (!command) return;
  
  if (command.template) {
    const texto = plantillas[command.template]
      .replace('[NOMBRE]', conversacionActiva.value.participante.nombre);
    setInputText(texto);
  } else if (command.action) {
    ejecutarAccion(command.action);
  }
};
```
