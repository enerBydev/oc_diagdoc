```sql
CREATE TABLE rendimiento_operadores (
  id UUID PRIMARY KEY,
  operador_id UUID REFERENCES users(id),
  periodo_inicio DATE NOT NULL,
  periodo_fin DATE NOT NULL,
  
  -- MÃ©tricas brutas
  total_servicios INTEGER DEFAULT 0,
  total_calificaciones INTEGER DEFAULT 0,
  suma_estrellas INTEGER DEFAULT 0,
  incidencias_count INTEGER DEFAULT 0,
  llegadas_tiempo INTEGER DEFAULT 0,
  
  -- Scores calculados
  score_csat DECIMAL(5,2),
  score_puntualidad DECIMAL(5,2),
  score_incidencias DECIMAL(5,2),
  score_productividad DECIMAL(5,2),
  score_total DECIMAL(5,2),
  
  -- ClasificaciÃ³n
  clasificacion VARCHAR(20),
  
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Vista para ranking
CREATE VIEW ranking_operadores AS
SELECT 
  u.nombre,
  r.score_total,
  r.clasificacion,
  RANK() OVER (ORDER BY r.score_total DESC) as posicion
FROM rendimiento_operadores r
JOIN users u ON r.operador_id = u.id
WHERE r.periodo_fin = date_trunc('month', now())::date;
```
